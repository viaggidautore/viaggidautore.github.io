<script>
const WEBHOOK = "https://n8n.ivandat.it/webhook/viaggi-autore";
const messagesContainer = document.getElementById('chatMessages');
const input = document.getElementById('chatInput');

// 1. MESSAGGIO INIZIALE
window.onload = () => {
  setTimeout(() => {
    addMsg("Ciao! Sono Marco. 🌍 Dove ti piacerebbe andare per il tuo prossimo viaggio?", false);
  }, 500);
};

// 2. INVIO MESSAGGIO
async function sendMessage() {
  const testo = input.value.trim();
  if (!testo) return;

  addMsg(testo, true);
  input.value = '';
  input.disabled = true;
  document.getElementById('chatSend').disabled = true;

  const typing = addTyping();

  try {
    const res = await fetch(WEBHOOK, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ messaggio: testo })
    });
    const data = await res.json();
    typing.remove();

    if (data.risposta) {
      addMsg(formatRisposta(data.risposta), false);
      if (data.risposta.includes('SI VOGLIO SERVIZIO')) showQuickReply();
    } else {
      addMsg('Si è verificato un errore. Riprova.', false);
    }
  } catch(e) {
    typing.remove();
    addMsg('Errore di connessione. Riprova.', false);
  }

  input.disabled = false;
  document.getElementById('chatSend').disabled = false;
  input.focus();
}

// 3. AGGIUNGI MESSAGGIO
function addMsg(testo, isUser) {
  const row = document.createElement('div');
  row.className = 'msg-row' + (isUser ? ' msg-user' : '');
  
  let html = '';
  if (!isUser) {
    html += `<div class="msg-av"><img src="https://images.unsplash.com/photo-1507003211169-0a1dd7228f2d?q=80&w=60&auto=format&fit=crop"></div>`;
  }
  
  // Aggiunta pulsante PDF se il messaggio è del bot e sembra un preventivo
  const pdfButton = (!isUser && (testo.includes('€') || testo.includes('Volo') || testo.includes('Itinerario'))) 
    ? `<button onclick="downloadPDF(this)" class="pdf-btn">📄 Scarica Preventivo (PDF)</button>` 
    : '';

  html += `<div class="msg-bubble">${testo}${pdfButton}</div>`;
  row.innerHTML = html;
  messagesContainer.appendChild(row);
  messagesContainer.scrollTop = messagesContainer.scrollHeight;
}

function addTyping() {
  const t = document.createElement('div');
  t.className = 'msg-row';
  t.innerHTML = `<div class="msg-typing"><span></span><span></span><span></span></div>`;
  messagesContainer.appendChild(t);
  messagesContainer.scrollTop = messagesContainer.scrollHeight;
  return t;
}

function formatRisposta(t) {
  return t.replace(/\n/g, '<br>').replace('SI VOGLIO SERVIZIO', '');
}

function showQuickReply() {
  document.getElementById('quickReplyWrap').style.display = 'block';
}

function handleQuickReply() {
  document.getElementById('quickReplyWrap').style.display = 'none';
  input.value = "SÌ, VOGLIO IL SERVIZIO COMPLETO! 🚀";
  sendMessage();
}

// Funzione INVIO con INVIO
input.addEventListener('keypress', (e) => { if (e.key === 'Enter') sendMessage(); });

// 4. GENERAZIONE PDF
function downloadPDF(btn) {
  const bubble = btn.parentElement;
  
  // Crea un clone per la stampa pulita
  const element = document.createElement('div');
  element.style.padding = '40px';
  element.style.fontFamily = 'Arial, sans-serif';
  element.style.color = '#0f1f35';
  
  // Header nel PDF
  const header = `
    <div style="text-align:center; margin-bottom: 30px; border-bottom: 2px solid #c8952a; padding-bottom: 10px;">
      <h1 style="margin:0; font-family: 'Playfair Display', serif;">Viaggi d'Autore</h1>
      <p style="margin:5px 0; color: #5a6070;">Il tuo preventivo personalizzato</p>
    </div>
  `;
  
  // Rimuove il pulsante dal testo del PDF
  let cleanText = bubble.innerHTML.replace(/<button.*<\/button>/, '');
  
  element.innerHTML = header + `<div style="line-height:1.6; font-size: 14px;">${cleanText}</div>`;

  const opt = {
    margin:       10,
    filename:     'Preventivo_Viaggio_Autore.pdf',
    image:        { type: 'jpeg', quality: 0.98 },
    html2canvas:  { scale: 2 },
    jsPDF:        { unit: 'mm', format: 'a4', orientation: 'portrait' }
  };

  html2pdf().from(element).set(opt).save();
}
