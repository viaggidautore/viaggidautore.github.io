<script>
// Webhook originale dal tuo file
const WEBHOOK = 'https://garnett-cleistogamic-unslowly.ngrok-free.dev/webhook/868a702c-ae44-4bbf-bcae-16d953de282e';

function addMsg(text, isUser) {
  const msgs = document.getElementById('chatMessages');
  const row = document.createElement('div');
  row.className = 'msg-row' + (isUser ? ' msg-user' : '');
  
  // Se è il bot e il testo sembra un preventivo, aggiungi il tasto PDF
  let btnHtml = "";
  if (!isUser && (text.includes('€') || text.includes('Volo') || text.includes('Hotel'))) {
    btnHtml = `<button class="pdf-btn" onclick="scaricaPDF(this)">📥 Scarica Preventivo PDF</button>`;
  }

  row.innerHTML = `<div class="msg-bubble">${text}${btnHtml}</div>`;
  msgs.appendChild(row);
  msgs.scrollTop = msgs.scrollHeight;
}

async function inviaMsg() {
  const input = document.getElementById('chatInput');
  const testo = input.value.trim();
  if (!testo) return;

  addMsg(testo, true);
  input.value = '';

  try {
    const res = await fetch(WEBHOOK, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ messaggio: testo })
    });
    const data = await res.json();
    if (data.risposta) {
      addMsg(data.risposta.replace(/\n/g, '<br>'), false);
    }
  } catch (e) {
    addMsg("Errore di connessione. Riprova.", false);
  }
}

function scaricaPDF(btn) {
  const elemento = btn.parentElement.cloneNode(true);
  const b = elemento.querySelector('.pdf-btn');
  if(b) b.remove(); // Non stampare il tasto nel PDF

  const opt = {
    margin: 1,
    filename: 'Preventivo_Viaggi_d_Autore.pdf',
    image: { type: 'jpeg', quality: 0.98 },
    html2canvas: { scale: 2 },
    jsPDF: { unit: 'in', format: 'letter', orientation: 'portrait' }
  };
  html2pdf().set(opt).from(elemento).save();
}

document.getElementById('chatInput').addEventListener('keydown', e => {
  if (e.key === 'Enter') inviaMsg();
});
