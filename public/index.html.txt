<script>
  const WEBHOOK = 'https://n8n.ivandat.it/webhook/viaggi-dautore-chat';

  // Contatore dinamico
  function initCounter() {
    const c = document.getElementById('today-counter');
    let val = 61 + Math.floor(Math.random() * 5);
    c.innerText = val;
    setInterval(() => { if(Math.random() > 0.8) { val++; c.innerText = val; } }, 15000);
  }
  initCounter();

  // Formattazione
  function formatRisposta(t) {
    t = t.replace(/\*\*(.*?)\*\*/g, '<b>$1</b>');
    t = t.replace(/\n/g, '<br>');
    t = t.replace(/•/g, '<br>•');
    return t;
  }

  function addMsg(text, isUser) {
    const container = document.getElementById('chatMsgs');
    const row = document.createElement('div');
    row.className = 'msg-row' + (isUser ? ' msg-user' : '');
    row.innerHTML = `<div class="msg-bubble">${text}</div>`;
    container.appendChild(row);
    container.scrollTop = container.scrollHeight;
  }

  function showQuickReply() {
    const area = document.getElementById('quickReplyArea');
    area.innerHTML = `<button class="quick-reply-btn" onclick="sendQuickReply('SI VOGLIO SERVIZIO')">✅ SI, VOGLIO IL SERVIZIO</button>`;
    area.style.display = 'flex';
  }

  function sendQuickReply(txt) {
    document.getElementById('quickReplyArea').style.display = 'none';
    document.getElementById('chatInput').value = txt;
    sendMsg();
  }

  async function sendMsg() {
    const input = document.getElementById('chatInput');
    const btn = document.getElementById('chatSend');
    const testo = input.value.trim();
    
    if (!testo || !puoInviarePreventivo()) return;

    addMsg(testo, true);
    input.value = '';
    input.disabled = true;
    btn.disabled = true;

    // Loader
    const loader = document.createElement('div');
    loader.className = 'loader-ricerca';
    loader.innerHTML = `<div class="spinner-small"></div><span>🔎 Sto cercando voli e hotel per te...</span>`;
    document.getElementById('chatMsgs').appendChild(loader);
    document.getElementById('chatMsgs').scrollTop = document.getElementById('chatMsgs').scrollHeight;

    try {
      const res = await fetch(WEBHOOK, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'ngrok-skip-browser-warning': 'true' },
        body: JSON.stringify({ messaggio: testo })
      });

      const data = await res.json();
      loader.remove();

      if (data.risposta) {
        addMsg(formatRisposta(data.risposta), false);
        if (data.risposta.toUpperCase().includes('SI VOGLIO SERVIZIO')) showQuickReply();
      }
    } catch (e) {
      if(loader) loader.remove();
      addMsg("Connessione lenta. Riprova tra un istante.", false);
    }

    input.disabled = false;
    btn.disabled = false;
    input.focus();
  }

  document.getElementById('chatInput').addEventListener('keypress', (e) => { if (e.key === 'Enter') sendMsg(); });
</script>
